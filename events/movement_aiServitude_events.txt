namespace = movement_aiServitude

# Called on monthly pulse
event = {
	id = movement_aiServitude.1
	
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		every_playable_country = {
			set_variable = { which = movement_aiServitude_support value = 0}
			set_variable = { which = movement_aiServitude_radicalism value = 0}
		
			if = {
				limit = {
					NOT = { has_ethic = ethic_gestalt_consciousness } OR = { can_set_ai_policy = yes has_country_flag = synthetic_empire } NOT = { has_country_flag = synthetic_empire }
					NOT = { has_policy_flag = ai_servitude }
				}
				set_country_flag = suppress_vanilla_factions
				country_event = { id = movement_aiServitude.2 }
			}
		}
	}
}

# Calculate support and radicalism by combining faction and numerical support
# Outputs:
# - country.movement_aiServitude_support
# - country.movement_aiServitude_radicalism
# - country.movement_aiServitude_inertia
country_event = {
	id = movement_aiServitude.2

	hide_window = yes
	is_triggered_only = yes

	immediate = {
		set_variable = { which = movement_aiServitude_faction_support value = 0 }
		set_variable = { which = movement_aiServitude_faction_radicalism value = 0 }
		set_variable = { which = movement_aiServitude_numerical_support value = 0 }
		set_variable = { which = movement_aiServitude_numerical_radicalism value = 0 }

		if = {
			limit = {
				NOT = { has_ethic = ethic_gestalt_consciousness } OR = { can_set_ai_policy = yes has_country_flag = synthetic_empire } NOT = { has_country_flag = synthetic_empire }
				
			}
			country_event = { id = movement_aiServitude.3 }
			country_event = { id = movement_aiServitude.4 }
			
			if = {
				limit = {
					check_variable = {which = movement_aiServitude_faction_support value > 0 }
					NOR = { 
						has_country_flag = movement_for_aiServitude  
						has_country_flag = aiServitude_revolt
					}
				}
				country_event = { id = movement_aiServitude.5 }
			}
		}		
		set_variable = { which = movement_aiServitude_support value = movement_aiServitude_faction_support }
		change_variable = { which = movement_aiServitude_support value = movement_aiServitude_numerical_support }
		divide_variable = { which = movement_aiServitude_support value = 2 }

		set_variable = { which = movement_aiServitude_radicalism value = movement_aiServitude_faction_radicalism }
		change_variable = { which = movement_aiServitude_radicalism value = movement_aiServitude_numerical_radicalism }
		divide_variable = { which = movement_aiServitude_radicalism value = 2 }

		set_variable = {which = movement_aiServitude_inertia value = movement_aiServitude_support}
		multiply_variable = { which = movement_aiServitude_inertia value = movement_aiServitude_radicalism }
	}
}

# Calculate support from factions
# Outputs:
# - country.movement_aiServitude_faction_support
# - country.movement_aiServitude_faction_radicalism
country_event = {
	id = movement_aiServitude.3
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		
		every_pop_faction = {
			
			if = {
				limit = {
					is_pop_faction_type = technologist
				}
				export_trigger_value_to_variable = { trigger = support variable = aiServitude_technologist_support }
				multiply_variable = {which = aiServitude_technologist_support value = 100}
				# aiServitude_technologist_support is 0 to 100
				
				export_trigger_value_to_variable = { trigger = faction_approval variable = aiServitude_technologist_approval }
				# aiServitude_technologist_approval is 0 to 1

				ROOT = {
					change_variable = { which = movement_aiServitude_faction_support value = PREV.aiServitude_technologist_support }
					
					set_variable = { which = aiServitude_technologist_radicalism value = 1 }
					subtract_variable = { which = aiServitude_technologist_radicalism value = PREV.aiServitude_technologist_approval }
					
					multiply_variable = { which = aiServitude_technologist_radicalism value = PREV.aiServitude_technologist_support }

					change_variable = { which = movement_aiServitude_faction_radicalism value = aiServitude_technologist_radicalism }
				}
			}
			
			if = {
				limit = {
					is_pop_faction_type = traditionalist
				}
				export_trigger_value_to_variable = { trigger = support variable = aiServitude_traditionalist_support }
				multiply_variable = {which = aiServitude_traditionalist_support value = 100}
				# aiServitude_traditionalist_support is 0 to 100
				
				export_trigger_value_to_variable = { trigger = faction_approval variable = aiServitude_traditionalist_approval }
				# aiServitude_traditionalist_approval is 0 to 1

				ROOT = {
					change_variable = { which = movement_aiServitude_faction_support value = PREV.aiServitude_traditionalist_support }
					
					set_variable = { which = aiServitude_traditionalist_radicalism value = 1 }
					subtract_variable = { which = aiServitude_traditionalist_radicalism value = PREV.aiServitude_traditionalist_approval }
					
					multiply_variable = { which = aiServitude_traditionalist_radicalism value = PREV.aiServitude_traditionalist_support }

					change_variable = { which = movement_aiServitude_faction_radicalism value = aiServitude_traditionalist_radicalism }
				}
			}
			
		}
	}
}

# Calculate support based on numbers
# Outputs:
# - country.movement_aiServitude_numerical_support
# - country.movement_aiServitude_numerical_radicalism
country_event = {
	id = movement_aiServitude.4
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		set_variable = { which = aiServitude_population value = 0}
		set_variable = { which = aiServitude_supporting_population value = 0}
		set_variable = { which = aiServitude_radical_population value = 0}
		
		every_owned_pop = {
			ROOT = {
				change_variable = { which = aiServitude_population value = 1}
			}
			
			if = {
				limit = {
					OR = {
						
						has_ethic = ethic_materialist
						has_ethic = ethic_fanatic_materialist
						
						has_ethic = ethic_spiritualist
						has_ethic = ethic_fanatic_spiritualist
						
					}
				}
				# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
				export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
				ROOT = {
					set_variable = {which = pop_aiServitude_radicalism value = 1}
					subtract_variable = { which = pop_aiServitude_radicalism value = PREV.pop_happiness }

					change_variable = { which = aiServitude_supporting_population value = 1 }
					change_variable = { which = aiServitude_radical_population value = pop_aiServitude_radicalism }
				}
			}
		}
		
		set_variable = { which = movement_aiServitude_numerical_support value = aiServitude_supporting_population}
		divide_variable = { which = movement_aiServitude_numerical_support value = aiServitude_population }
		multiply_variable = {which = movement_aiServitude_numerical_support value = 100}

		set_variable = { which = movement_aiServitude_numerical_radicalism value = aiServitude_radical_population}
		divide_variable = { which = movement_aiServitude_numerical_radicalism value = aiServitude_population }
		multiply_variable = {which = movement_aiServitude_numerical_radicalism value = 100}
	}
}

		}
	}
}

# Create a revolution
situation_event = {
	id = movement_aiServitude.6
	title = movement_aiServitude.6.title
	desc = movement_aiServitude.6.desc
	picture = GFX_evt_night_raid
	is_triggered_only = yes
	hide_window = no

	immediate = {
		set_variable = { which = aiServitude_largest_planet_inertia value = 0}
		OWNER = {
			set_country_flag = aiServitude_revolt
			every_owned_planet = {
				set_variable = { which = aiServitude_planet_inertia value = 0}
				
				every_owned_pop = {
					
					# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
					export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
					change_variable = { which = pop_happiness value = 0.5 }

					export_modifier_to_variable = { modifier = pop_political_power variable = pop_political_power }
					change_variable = { which = pop_political_power value = 0.5 }
					
					set_variable = {which = pop_aiServitude_radicalism value = 1}
					subtract_variable = { which = pop_aiServitude_radicalism value = pop_happiness }
					
					set_variable = { which = aiServitude_pop_inertia value = pop_political_power }
					multiply_variable = { which = aiServitude_pop_inertia value = pop_aiServitude_radicalism }
					log = "Planet [Prev.GetName] - Pop Happiness: [This.pop_happiness] Political Power: [This.pop_political_power] Inertia: [This.aiServitude_pop_inertia]"

					if = {
						limit = {
							NOR = {
								
								has_ethic = ethic_materialist
								has_ethic = ethic_fanatic_materialist
								
								has_ethic = ethic_spiritualist
								has_ethic = ethic_fanatic_spiritualist
								
							}
						}
						multiply_variable = { which = aiServitude_pop_inertia value = -1 }
					}
					PREV = {
						change_variable = { which = aiServitude_planet_inertia value = PREV.aiServitude_pop_inertia }
					}
				}
				log = "Total Planet [This.GetName] revolt inertia: [This.aiServitude_planet_inertia]"
				if = {
					limit = {
						check_variable = { which = aiServitude_planet_inertia value > ROOT.aiServitude_largest_planet_inertia }
					}
					save_event_target_as = aiServitude_revolting_capital
					ROOT = {
						set_variable = { which = aiServitude_largest_planet_inertia value = PREV.aiServitude_planet_inertia }
					}
				}
			}
		}
		log = "Revolt Chosen Capital: [event_target:aiServitude_revolting_capital.GetName]"
		event_target:aiServitude_revolting_capital = {
			random_owned_species = {
				save_event_target_as = aiServitude_revolting_species
			}
			create_rebels = {
				name = random
				ethos = random
				authority = random
				civics = random
				origin = "origin_separatists"
				species = event_target:aiServitude_revolting_species
				flag = random
				effect = {
					save_event_target_as = rebel_empire
					set_country_type = default
					establish_communications_no_message = root.owner
		
					event_target:aiServitude_revolting_capital = { #Flip the target planet
						set_owner = event_target:rebel_empire
			
						solar_system = {
							if = {
								limit = { exists = starbase }
								starbase = {
									if = {
										limit = {
											is_owned_by = root.owner
										}
										set_owner = event_target:rebel_empire
										if = {
											limit = { has_starbase_size = starbase_outpost }
											set_starbase_size = starbase_starport
											set_starbase_module = {
												slot = 1
												module = shipyard
											}
											set_starbase_module = {
												slot = 2
												module = shipyard
											}
										}
										else_if = {
											limit = {
												NOT = { has_starbase_module = shipyard }
											}
											set_starbase_module = {
												slot = 1
												module = shipyard
											}
											set_starbase_module = {
												slot = 2
												module = shipyard
											}
										}
									}
								}
							}
							else = {
								create_starbase = {
									size = starbase_starport
									module = shipyard
									module = shipyard
									owner = event_target:rebel_empire
								}
							}
							add_claims = {
								who = root.owner
								show_notification = no
							}

							every_system_planet = {
								limit = {
									is_colony = yes
									NOT = { is_same_value = event_target:rebel_planet }
									is_owned_by = root.owner
								}
								set_owner = event_target:rebel_empire
							}
							every_fleet_in_system = {
								limit = {
									is_owned_by = root.owner
									can_go_mia = yes
								}
								set_mia = mia_return_home
							}
						}
					}
					set_policy = { 
						policy = artificial_intelligence_policy 
						option = ai_servitude 
						cooldown = yes 
					}
					add_resource = {
						minerals = 15000
						energy = 15000
						food = 15000
						consumer_goods = 10000
						alloys = 10000
						rare_crystals = 1000
						volatile_motes = 1000
						exotic_gases = 1000
						influence = 1000
						unity = 10000
					}

					root.owner = {
						every_owned_planet = { #Flip the additional joiner planets
							limit = {
								check_variable = { which = aiServitude_planet_inertia value > 0}
								
								OR = {
									NOT = { exists = starbase }
									starbase = { is_owned_by = root.owner }
								}
								has_ground_combat = no
								has_orbital_bombardment = no
							}
							if = { #if you check this in the original limit of every_owned_planet, it calcs it before the loop is executed. Which means it will set the owner of the system twice if there are two colonies in the system.
								limit = {
									exists = controller
									controller = { is_same_value = root.owner }
								}
								set_variable = { which = aiServitude_planet_inertia value = 0}
								set_owner = event_target:rebel_empire
		
								solar_system = {
									if = {
										limit = { exists = starbase }
										starbase = { set_owner = event_target:rebel_empire }
									}
									add_claims = {
										who = root.owner
										show_notification = no
									}
									every_system_planet = {
										limit = {
											is_colony = yes
											NOT = { is_same_value = prevprev }
											is_owned_by = root.owner
										}
										set_owner = event_target:rebel_empire
									}
									every_fleet_in_system = {
										limit = {
											is_owned_by = root.owner
											can_go_mia = yes
										}
										set_mia = mia_return_home
									}
								}
							}
						}
					}
					every_system_within_border = { #Add a few systems
						every_neighbor_system = {
							limit = {
								exists = owner
								owner = { is_same_value = root.owner }
								exists = starbase
								NOR = {
									starbase = {
										OR = {
											is_starbase_type = sfortress
											NOT = { is_owned_by = root.owner }
										}
									}
									any_system_planet = {
										is_colony = yes
									}
								}
							}
							starbase = { set_owner = event_target:rebel_empire }
							add_claims = {
								who = root.owner
								show_notification = no
							}
							every_fleet_in_system = {
								limit = {
									is_owned_by = root.owner
									can_go_mia = yes
								}
								set_mia = mia_return_home
							}
						}
					}
					while = { #Get rid of bordergore systems
						limit = { #Note: system_within_border caching is quite funky at a time like this when many borders are changing hands. The starbase ownership is checked to be extra sure, since it is not cached (unlike system ownership - though that probably would have worked too)
							root.owner = {
								any_system_within_border = {
									exists = starbase
									starbase.owner = { is_same_value = prevprev }
									count_neighbor_system = {
										count >= 2
										limit = {
											exists = starbase.owner
											starbase.owner = {
												is_same_value = event_target:rebel_empire
											}
										}
									}
									NOR = {
										starbase = { is_starbase_type = sfortress }
										any_system_planet = {
											is_colony = yes
										}
									}
								}
							}
						}
						root.owner = {
							every_system_within_border = {
								limit = {
									exists = starbase
									starbase.owner = { is_same_value = prevprev }
									count_neighbor_system = {
										count >= 2
										limit = {
											exists = starbase.owner
											starbase.owner = {
												is_same_value = event_target:rebel_empire
											}
										}
									}
									NOR = {
										starbase = { is_starbase_type = sfortress }
										any_system_planet = {
											is_colony = yes
										}
									}
								}
								starbase = {
									set_owner = event_target:rebel_empire
								}
								add_claims = {
									who = root.owner
									show_notification = no
								}
								every_fleet_in_system = {
									limit = {
										is_owned_by = root.owner
										can_go_mia = yes
									}
									set_mia = mia_return_home
								}
							}
						}
					}	
					
					shift_ethic = ethic_materialist
					
					shift_ethic = ethic_spiritualist
					
					
					change_government = {
						authority = random
						civics = random
					}
					create_leader = {
						class = random_ruler
						species = owner_main_species
						name = random
						skill = 5
						effect = {
							event_target:rebel_empire = { assign_leader = prev }
							# add_random_leader_trait = yes
							# add_random_leader_trait = yes
						}
					}
					
					create_fleet_from_naval_cap = 2
					declare_war = {
						target = Root.Owner
						name = "Civil War over AI Servitude"
						attacker_war_goal = wg_conquest
					}
				}
			}
		}
		destroy_situation = this
	}
	option = {
		name = movement_aiServitude.6.fight
	}
	option = {
		name = movement_aiServitude.6.join
		event_target:rebel_empire = {
			set_player = root.owner
		}
	}
}
