# Determine advance fail or stall
situation_event = {
	id = situation_passPolicy_defensiveWarsOnly.2	
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		if = {
			limit = {
				NOT = {
					is_variable_set = setback_count
				}
			}
			set_variable = { which = setback_count value = 0}
		}
		if = {
			limit = {
				NOT = {
					is_variable_set = swayed_to_support
				}
			}
			set_variable = { which = swayed_to_support value = 0}
		}
		if = {
			limit = {
				NOT = {
					is_variable_set = swayed_to_detract
				}
			}
			set_variable = { which = swayed_to_detract value = 0}
		}
		set_variable = { which = faction_support value = 0}
		set_variable = { which = faction_detraction value = 0}
	
		OWNER = {
			every_pop_faction = {
				
				if = {
					limit = {
						is_pop_faction_type = isolationist
					}
					export_trigger_value_to_variable = { trigger = support variable = isolationist_support }
					multiply_variable = {which = isolationist_support value = 100}
					ROOT = {
						log = "[PREV.isolationist_support]"
						change_variable = { which = faction_support value = PREV.isolationist_support }		
					}
				}
				
				if = {
					limit = {
						is_pop_faction_type = prosperity
					}
					export_trigger_value_to_variable = { trigger = support variable = prosperity_support }
					multiply_variable = {which = prosperity_support value = 100}
					ROOT = {
						log = "[PREV.prosperity_support]"
						change_variable = { which = faction_support value = PREV.prosperity_support }		
					}
				}
				
				
				if = {
					limit = {
						is_pop_faction_type = imperialist
					}
					export_trigger_value_to_variable = { trigger = support variable = imperialist_support }
					multiply_variable = {which = imperialist_support value = 100}
					ROOT = {
						change_variable = { which = faction_detraction value = PREV.imperialist_support }	
					}
				}
				
				if = {
					limit = {
						is_pop_faction_type = supremacist
					}
					export_trigger_value_to_variable = { trigger = support variable = supremacist_support }
					multiply_variable = {which = supremacist_support value = 100}
					ROOT = {
						change_variable = { which = faction_detraction value = PREV.supremacist_support }	
					}
				}
				
				if = {
					limit = {
						is_pop_faction_type = xenoist
					}
					export_trigger_value_to_variable = { trigger = support variable = xenoist_support }
					multiply_variable = {which = xenoist_support value = 100}
					ROOT = {
						change_variable = { which = faction_detraction value = PREV.xenoist_support }	
					}
				}
				
			}
		}

		set_variable = {which = faction_ambivalence value = 100}
		subtract_variable = { which = faction_ambivalence value = faction_support }
		subtract_variable = { which = faction_ambivalence value = faction_detraction }
		
		log = "PRE MATH: defensiveWarsOnly support: [THIS.faction_support]+[THIS.swayed_to_support] detraction: [THIS.faction_detraction]+[THIS.swayed_to_detract]"	

		# If swayed is positive, give it support, if negative, give it to detraction
		
		change_variable = { which = faction_support value = swayed_to_support }
		change_variable = { which = faction_detraction value = swayed_to_detract }
		

		# Clamp variables
		if = {
			limit = {
				check_variable = { which = faction_support value > 100}
			}
			set_variable = { which = faction_support value = 100}
		}
		if = {
			limit = {
				check_variable = { which = faction_support value < 0}
			}
			set_variable = { which = faction_support value = 0}
		}
		
		if = {
			limit = {
				check_variable = { which = faction_detraction value > 100}
			}
			set_variable = { which = faction_detraction value = 100}
		}
		if = {
			limit = {
				check_variable = { which = faction_detraction value < 0}
			}
			set_variable = { which = faction_detraction value = 0}
		}

		set_variable = {which = faction_ambivalence value = 100}
		subtract_variable = { which = faction_ambivalence value = faction_support }
		subtract_variable = { which = faction_ambivalence value = faction_detraction }

		if = {
			limit = {
				check_variable = { which = faction_ambivalence value > 100}
			}
			set_variable = { which = faction_ambivalence value = 100}
		}
		if = {
			limit = {
				check_variable = { which = faction_ambivalence value < 0}
			}
			set_variable = { which = faction_ambivalence value = 0}
		}


		ceiling_variable = faction_support
		ceiling_variable = faction_detraction
		ceiling_variable = faction_ambivalence
		log = "POST MATH: defensiveWarsOnly support: [THIS.faction_support] detraction: [THIS.faction_detraction] ambivalence: [THIS.faction_ambivalence]"	
	}
}
# Advance or not
situation_event = {
	id = situation_passPolicy_defensiveWarsOnly.3
	is_triggered_only = yes
	hide_window = yes
	
	immediate = {
		# Update all values
		situation_event = {
			id = situation_passPolicy_defensiveWarsOnly.2	
		}
		set_variable_to_random_value = { which = advancement_roll min = 0 max = 100 rounded = yes}
		log = "Rolled: [THIS.advancement_roll] [THIS.faction_support]:[THIS.faction_detraction]:[THIS.faction_ambivalence]"
		if = {
			limit = {
				check_variable = { which = advancement_roll value < faction_support  }
			}
			# Advance!
			situation_event = { id = situation_passPolicy_defensiveWarsOnly.4}
		}
		else = {
			subtract_variable = { which = advancement_roll value = faction_support }
			if = {
				limit = {
					check_variable = { which = advancement_roll value < faction_detraction }
				}
				# Fail!
				situation_event = {id = situation_passPolicy_defensiveWarsOnly.5}
			}
			else = {
				# Ambivalence!
				situation_event = {id = situation_passPolicy_defensiveWarsOnly.6}
			}
		}	
	}
}
# Advance

	}
}
# Fail
situation_event = {
	id = situation_passPolicy_defensiveWarsOnly.5
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		change_variable = { which = setback_count value = 1 }
		log = "defensiveWarsOnly fail! [this.setback_count]/3"
		if = {
			limit = { check_variable = { which = setback_count value = 3}}
			log = "defensiveWarsOnly failed due to setbacks [this.setback_count]/3!"
			situation_event = { id = situation_passPolicy_defensiveWarsOnly.9}
		}
	}
}

# Ambivalent
situation_event = {
	id = situation_passPolicy_defensiveWarsOnly.6
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		if = {
			limit = {
				check_variable = { which = faction_ambivalence value > 10}
			}
			set_variable_to_random_value = { which = sway_roll min = 0 max = 100 rounded = yes}
			if = {
				limit = {
					check_variable = { which = sway_roll value > 50 }
				}
				log = "defensiveWarsOnly ambivalent! Rolled [THIS.sway_roll]. Giving 10 ambivalence support."
				change_variable = { which = swayed_to_support value = 10}
			}
			else = {
				log = "defensiveWarsOnly ambivalent! Rolled [THIS.sway_roll]. Giving 10 ambivalence detraction."
				change_variable = { which = swayed_to_detract value = 10}
			}
			situation_event = {id = situation_passPolicy_defensiveWarsOnly.2 }
		}
		else = {
			log = "defensiveWarsOnly ambivalent! No ambivalence to give at [THIS.faction_ambivalence]."
		}
	}
}
