	id = movement_defensiveWarsOnly.1
	
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		every_playable_country = {
			set_variable = { which = movement_defensiveWarsOnly_support value = 0}

				limit = {
					NOR = { has_ethic = ethic_fanatic_militarist has_valid_civic = civic_fanatic_purifiers has_valid_civic = civic_hive_devouring_swarm has_valid_civic = civic_machine_terminator has_civic = civic_crusader_spirit }
					NOT = { has_policy_flag = no_wars }
				}
				country_event = { id = movement_defensiveWarsOnly.2 }
			}
		}
	}
}

# Calculate support from factions
# Outputs:
# - country.movement_defensiveWarsOnly_faction_support
# - country.movement_defensiveWarsOnly_faction_radicalism
country_event = {
	id = movement_defensiveWarsOnly.3
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		
		every_pop_faction = {
			
			if = {
				limit = {
					is_pop_faction_type = isolationist
				}
				export_trigger_value_to_variable = { trigger = support variable = defensiveWarsOnly_faction_support }
				multiply_variable = {which = defensiveWarsOnly_faction_support value = 100}
				# defensiveWarsOnly_faction_support is 0 to 100
				
				export_modifier_to_variable = { modifier = faction_approval variable = defensiveWarsOnly_faction_approval }
				subtract_variable = {which = defensiveWarsOnly_faction_approval value = 0.5}
				multiply_variable = {which = defensiveWarsOnly_faction_approval value = 2}
				# defensiveWarsOnly_faction_approval is -1 to 1
				
				multiply_variable = { which = defensiveWarsOnly_faction_approval value=defensiveWarsOnly_faction_support }

				ROOT = {
					change_variable = { which = movement_defensiveWarsOnly_faction_support value = PREV.defensiveWarsOnly_faction_support }
					if = {
						limit = { 
							check_variable = { which = PREV.defensiveWarsOnly_faction_approval value < 0}
						}
						change_variable = { which = movement_defensiveWarsOnly_faction_radicalism value = PREV.defensiveWarsOnly_faction_support }
					}
				}
			}
			
			if = {
				limit = {
					is_pop_faction_type = prosperity
				}
				export_trigger_value_to_variable = { trigger = support variable = defensiveWarsOnly_faction_support }
				multiply_variable = {which = defensiveWarsOnly_faction_support value = 100}
				# defensiveWarsOnly_faction_support is 0 to 100
				
				export_modifier_to_variable = { modifier = faction_approval variable = defensiveWarsOnly_faction_approval }
				subtract_variable = {which = defensiveWarsOnly_faction_approval value = 0.5}
				multiply_variable = {which = defensiveWarsOnly_faction_approval value = 2}
				# defensiveWarsOnly_faction_approval is -1 to 1
				
				multiply_variable = { which = defensiveWarsOnly_faction_approval value=defensiveWarsOnly_faction_support }

				ROOT = {
					change_variable = { which = movement_defensiveWarsOnly_faction_support value = PREV.defensiveWarsOnly_faction_support }
					if = {
						limit = { 
							check_variable = { which = PREV.defensiveWarsOnly_faction_approval value < 0}
						}
						change_variable = { which = movement_defensiveWarsOnly_faction_radicalism value = PREV.defensiveWarsOnly_faction_support }
					}
				}
			}
			
		}
	}
}

# Calculate support based on numbers
# Outputs:
# - country.movement_defensiveWarsOnly_numerical_support
country_event = {
	id = movement_defensiveWarsOnly.4
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		set_variable = { which = defensiveWarsOnly_population value = 0}
		set_variable = { which = defensiveWarsOnly_supporting_population value = 0}
		
		every_owned_pop = {
			ROOT = {
				change_variable = { which = defensiveWarsOnly_population value = 1}
			}
			
			if = {
				limit = {
					OR = {
						
						has_ethic = ethic_xenophobe
						has_ethic = ethic_fanatic_xenophobe
						
						has_ethic = ethic_pacifist
						has_ethic = ethic_fanatic_pacifist
						
					}
				}
				# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 numerical support whereas his happy cousin contributes none.
				export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
				set_variable = {which = pop_defensiveWarsOnly_support value = 1}
				subtract_variable = { which = pop_defensiveWarsOnly_support value = pop_happiness }
				ROOT = {
					change_variable = { which = defensiveWarsOnly_supporting_population value = PREV.pop_defensiveWarsOnly_support }
				}
			}
		}
		
		set_variable = { which = movement_defensiveWarsOnly_numerical_support value = defensiveWarsOnly_supporting_population}
		divide_variable = { which = movement_defensiveWarsOnly_numerical_support value = defensiveWarsOnly_population }
	}
}
