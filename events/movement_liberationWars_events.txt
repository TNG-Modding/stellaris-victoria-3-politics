# Called on monthly pulse
event = {
	id = movement_liberationWars.1
	
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		every_playable_country = {
			set_variable = { which = movement_liberationWars_support value = 0}
			set_variable = { which = movement_liberationWars_radicalism value = 0}
		
			if = {
				limit = {
					NOR = {has_ethic = ethic_fanatic_pacifist has_valid_civic = civic_fanatic_purifiers}
					NOT = { has_policy_flag = liberation_wars }
				}
				country_event = { id = movement_liberationWars.2 }
			}
		}
	}
}

# Calculate support from factions
# Outputs:
# - country.movement_liberationWars_faction_support
# - country.movement_liberationWars_faction_radicalism
country_event = {
	id = movement_liberationWars.3
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		
		every_pop_faction = {
			
			if = {
				limit = {
					is_pop_faction_type = imperialist
				}
				export_trigger_value_to_variable = { trigger = support variable = liberationWars_faction_support }
				multiply_variable = {which = liberationWars_faction_support value = 100}
				# liberationWars_faction_support is 0 to 100
				
				export_modifier_to_variable = { modifier = faction_approval variable = liberationWars_faction_approval }
				subtract_variable = {which = liberationWars_faction_approval value = 0.5}
				multiply_variable = {which = liberationWars_faction_approval value = 2}
				# liberationWars_faction_approval is -1 to 1
				
				multiply_variable = { which = liberationWars_faction_approval value=liberationWars_faction_support }

				ROOT = {
					change_variable = { which = movement_liberationWars_faction_support value = PREV.liberationWars_faction_support }
					if = {
						limit = { 
							check_variable = { which = PREV.liberationWars_faction_approval value < 0}
						}
						change_variable = { which = movement_liberationWars_faction_radicalism value = PREV.liberationWars_faction_support }
					}
				}
			}
			
			if = {
				limit = {
					is_pop_faction_type = xenoist
				}
				export_trigger_value_to_variable = { trigger = support variable = liberationWars_faction_support }
				multiply_variable = {which = liberationWars_faction_support value = 100}
				# liberationWars_faction_support is 0 to 100
				
				export_modifier_to_variable = { modifier = faction_approval variable = liberationWars_faction_approval }
				subtract_variable = {which = liberationWars_faction_approval value = 0.5}
				multiply_variable = {which = liberationWars_faction_approval value = 2}
				# liberationWars_faction_approval is -1 to 1
				
				multiply_variable = { which = liberationWars_faction_approval value=liberationWars_faction_support }

				ROOT = {
					change_variable = { which = movement_liberationWars_faction_support value = PREV.liberationWars_faction_support }
					if = {
						limit = { 
							check_variable = { which = PREV.liberationWars_faction_approval value < 0}
						}
						change_variable = { which = movement_liberationWars_faction_radicalism value = PREV.liberationWars_faction_support }
					}
				}
			}
			
		}
	}
}

# Calculate support based on numbers
# Outputs:
# - country.movement_liberationWars_numerical_support
country_event = {
	id = movement_liberationWars.4
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		set_variable = { which = liberationWars_population value = 0}
		set_variable = { which = liberationWars_supporting_population value = 0}
		
		every_owned_pop = {
			ROOT = {
				change_variable = { which = liberationWars_population value = 1}
			}
			
			if = {
				limit = {
					OR = {
						
						has_ethic = ethic_militarist
						has_ethic = ethic_fanatic_militarist
						
						has_ethic = ethic_xenophile
						has_ethic = ethic_fanatic_xenophile
						
					}
				}
				# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 numerical support whereas his happy cousin contributes none.
				export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
				set_variable = {which = pop_liberationWars_support value = 1}
				subtract_variable = { which = pop_liberationWars_support value = pop_happiness }
				ROOT = {
					change_variable = { which = liberationWars_supporting_population value = PREV.pop_liberationWars_support }
				}
			}
		}
		
		set_variable = { which = movement_liberationWars_numerical_support value = liberationWars_supporting_population}
		divide_variable = { which = movement_liberationWars_numerical_support value = liberationWars_population }
	}
}
