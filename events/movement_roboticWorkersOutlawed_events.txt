			}
		}
	}
}

# Calculate support from factions
# Outputs:
# - country.movement_roboticWorkersOutlawed_faction_support
# - country.movement_roboticWorkersOutlawed_faction_radicalism
country_event = {
	id = movement_roboticWorkersOutlawed.3
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		
		every_pop_faction = {
			
			if = {
				limit = {
					is_pop_faction_type = traditionalist
				}
				export_trigger_value_to_variable = { trigger = support variable = roboticWorkersOutlawed_faction_support }
				multiply_variable = {which = roboticWorkersOutlawed_faction_support value = 100}
				# roboticWorkersOutlawed_faction_support is 0 to 100
				
				export_modifier_to_variable = { modifier = faction_approval variable = roboticWorkersOutlawed_faction_approval }
				subtract_variable = {which = roboticWorkersOutlawed_faction_approval value = 0.5}
				multiply_variable = {which = roboticWorkersOutlawed_faction_approval value = 2}
				# roboticWorkersOutlawed_faction_approval is -1 to 1
				
				multiply_variable = { which = roboticWorkersOutlawed_faction_approval value=roboticWorkersOutlawed_faction_support }

				ROOT = {
					change_variable = { which = movement_roboticWorkersOutlawed_faction_support value = PREV.roboticWorkersOutlawed_faction_support }
					if = {
						limit = { 
							check_variable = { which = PREV.roboticWorkersOutlawed_faction_approval value < 0}
						}
						change_variable = { which = movement_roboticWorkersOutlawed_faction_radicalism value = PREV.roboticWorkersOutlawed_faction_support }
					}
				}
			}
			
		}
	}
}

# Calculate support based on numbers
# Outputs:
# - country.movement_roboticWorkersOutlawed_numerical_support
country_event = {
	id = movement_roboticWorkersOutlawed.4
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		set_variable = { which = roboticWorkersOutlawed_population value = 0}
		set_variable = { which = roboticWorkersOutlawed_supporting_population value = 0}
		
		every_owned_pop = {
			ROOT = {
				change_variable = { which = roboticWorkersOutlawed_population value = 1}
			}
			
			if = {
				limit = {
					OR = {
						
						has_ethic = ethic_spiritualist
						has_ethic = ethic_fanatic_spiritualist
						
					}
				}
				# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 numerical support whereas his happy cousin contributes none.
				export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
				set_variable = {which = pop_roboticWorkersOutlawed_support value = 1}
				subtract_variable = { which = pop_roboticWorkersOutlawed_support value = pop_happiness }
				ROOT = {
					change_variable = { which = roboticWorkersOutlawed_supporting_population value = PREV.pop_roboticWorkersOutlawed_support }
				}
			}
		}
		
		set_variable = { which = movement_roboticWorkersOutlawed_numerical_support value = roboticWorkersOutlawed_supporting_population}
		divide_variable = { which = movement_roboticWorkersOutlawed_numerical_support value = roboticWorkersOutlawed_population }
	}
}
