# Calculate support from factions
# Outputs:
# - country.movement_{{varname}}_faction_support
# - country.movement_{{varname}}_faction_radicalism
country_event = {
	id = movement_{{varname}}.3
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		
		every_pop_faction = {
			{% for supportingFaction in supportingFactions %}
			if = {
				limit = {
					is_pop_faction_type = {{supportingFaction}}
				}
				export_trigger_value_to_variable = { trigger = support variable = {{varname}}_faction_support }
				multiply_variable = {which = {{varname}}_faction_support value = 100}
				# {{varname}}_faction_support is 0 to 100
				
				export_modifier_to_variable = { modifier = faction_approval variable = {{varname}}_faction_approval }
				subtract_variable = {which = {{varname}}_faction_approval value = 0.5}
				multiply_variable = {which = {{varname}}_faction_approval value = 2}
				# {{varname}}_faction_approval is -1 to 1
				
				multiply_variable = { which = {{varname}}_faction_approval value={{varname}}_faction_support }

				ROOT = {
					change_variable = { which = movement_{{varname}}_faction_support value = PREV.{{varname}}_faction_support }
					if = {
						limit = { 
							check_variable = { which = PREV.{{varname}}_faction_approval value < 0}
						}
						change_variable = { which = movement_{{varname}}_faction_radicalism value = PREV.{{varname}}_faction_support }
					}
				}
			}
			{% endfor %}
		}
	}
}

# Calculate support based on numbers
# Outputs:
# - country.movement_{{varname}}_numerical_support
country_event = {
	id = movement_{{varname}}.4
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		set_variable = { which = {{varname}}_population value = 0}
		set_variable = { which = {{varname}}_supporting_population value = 0}
		
		every_owned_pop = {
			ROOT = {
				change_variable = { which = {{varname}}_population value = 1}
			}
			
			if = {
				limit = {
					OR = {
						{% for supportingEthic in supportingEthics%}
						has_ethic = ethic_{{supportingEthic}}
						has_ethic = ethic_fanatic_{{supportingEthic}}
						{% endfor %}
					}
				}
				# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 numerical support whereas his happy cousin contributes none.
				export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
				set_variable = {which = pop_{{varname}}_support value = 1}
				subtract_variable = { which = pop_{{varname}}_support value = pop_happiness }
				ROOT = {
					change_variable = { which = {{varname}}_supporting_population value = PREV.pop_{{varname}}_support }
				}
			}
		}
		
		set_variable = { which = movement_{{varname}}_numerical_support value = {{varname}}_supporting_population}
		divide_variable = { which = movement_{{varname}}_numerical_support value = {{varname}}_population }
	}
}
