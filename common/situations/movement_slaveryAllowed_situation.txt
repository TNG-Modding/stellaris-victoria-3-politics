situation_movement_slaveryAllowed = {
	picture = GFX_evt_broken_shackles
	complete_icon_frame = GFX_situation_outcome_frame_red
	complete_icon = GFX_situation_outcome_negative

	on_start = { 
		set_situation_approach = movement_approach.1
		# situation_event = { id = situation_movement_slaveryAllowed.2 } 
		OWNER = {
			set_country_flag = movement_for_slaveryAllowed
		}
	}

	monthly_progress = {
		base = 0
		modifier = {
			desc = movement_slaveryAllowed_gaining_revolutionary_progress
			OWNER = {
				check_variable = { which = movement_slaveryAllowed_radicalism value > 25}
			}
			add = 1
		}
		modifier = {
			desc = movement_slaveryAllowed_gaining_revolutionary_progress
			OWNER = {
				check_variable = { which = movement_slaveryAllowed_radicalism value < 25}
			}
			situation_progress > 0
			add = -1
		}
		modifier = {
			desc = movement_satisfied_with_passing_law
			OWNER = {
				has_country_flag = enacting_slaveryAllowed
			}
			situation_progress > 0
			add = -1
		}
		
	}

	on_progress_complete = {
		custom_tooltip = situation_movement_finisher_slaveryAllowed
		situation_event = { id = movement_slaveryAllowed.6 }
		
		OWNER = {
			remove_country_flag = movement_for_slaveryAllowed
		}
	}
	on_abort = {
		OWNER = {
			remove_country_flag = movement_for_slaveryAllowed
		}
	}

	stages = {
		stage_1 = {
			icon = GFX_situation_stage_1

		# }
	}
	monthly_progress = {
		base = 0
	}
	abort_trigger = {
		NOT = {
			target.owner = { NOT = { has_ethic = ethic_gestalt_consciousness } NOR = { any_agreement = { agreement_preset = preset_signatory } has_origin = origin_broken_shackles } OR = { has_ethic = ethic_authoritarian has_ethic = ethic_fanatic_authoritarian has_ethic = ethic_xenophobe has_ethic = ethic_fanatic_xenophobe } }
		}
