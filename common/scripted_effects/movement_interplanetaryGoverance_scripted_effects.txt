
calculate_interplanetaryGoverance_movement_support_and_radicalism_from_factions = {
	every_pop_faction = {
		
		if = {
			limit = {
				OR = {
					
					is_pop_faction_type = consolidationists
					
					
					
					
						
					
					
						AND = {
							is_pop_faction_type = consolidationists
							ROOT = { has_policy_flag = dependentColonies }
						}
					
					
				}
				
				NAND = {
					is_pop_faction_type = preFtlConservatives
					ROOT = { has_policy_flag = dependentColonies }
				}
				
				NAND = {
					is_pop_faction_type = colonists
					ROOT = { has_policy_flag = dependentColonies }
				}
				
			}
		
			export_trigger_value_to_variable = { trigger = support variable = interplanetaryGoverance_support }
			multiply_variable = {which = interplanetaryGoverance_support value = 100}
			# interplanetaryGoverance_support is 0 to 100
			
			export_trigger_value_to_variable = { trigger = faction_approval variable = interplanetaryGoverance_approval }
			# interplanetaryGoverance_approval is 0 to 1

			ROOT = {
				change_variable = { which = $SUPPORT$ value = PREV.interplanetaryGoverance_support }
				
				set_variable = { which = interplanetaryGoverance_radicalism value = 1 }
				subtract_variable = { which = interplanetaryGoverance_radicalism value = PREV.interplanetaryGoverance_approval }
				
				multiply_variable = { which = interplanetaryGoverance_radicalism value = PREV.interplanetaryGoverance_support }

				change_variable = { which = $RADICALISM$ value = interplanetaryGoverance_radicalism }
			}
		}
	}
}

calculate_interplanetaryGoverance_movement_support_and_radical_from_numbers = {
	set_variable = { which = interplanetaryGoverance_population value = 0}
	set_variable = { which = interplanetaryGoverance_supporting_population value = 0}
	set_variable = { which = interplanetaryGoverance_radical_population value = 0}
	
	every_owned_pop = {
		ROOT = {
			change_variable = { which = interplanetaryGoverance_population value = 1}
		}
		
		if = {
			limit = {
				OR = {
					
					has_ethic = ethic_authoritarian
					has_ethic = ethic_fanatic_authoritarian
					
				}
			}
			# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
			export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
			ROOT = {
				set_variable = {which = pop_interplanetaryGoverance_radicalism value = 1}
				subtract_variable = { which = pop_interplanetaryGoverance_radicalism value = PREV.pop_happiness }

				change_variable = { which = interplanetaryGoverance_supporting_population value = 1 }
				change_variable = { which = interplanetaryGoverance_radical_population value = pop_interplanetaryGoverance_radicalism }
			}
		}
	}
	
	set_variable = { which = $SUPPORT$ value = interplanetaryGoverance_supporting_population}
	divide_variable = { which = $SUPPORT$ value = interplanetaryGoverance_population }
	multiply_variable = {which = $SUPPORT$ value = 100}

	set_variable = { which = $RADICALISM$ value = interplanetaryGoverance_radical_population}
	divide_variable = { which = $RADICALISM$ value = interplanetaryGoverance_population }
	multiply_variable = {which = $RADICALISM$ value = 100}
}
