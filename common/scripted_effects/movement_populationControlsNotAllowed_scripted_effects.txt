
calculate_populationControlsNotAllowed_movement_support_and_radicalism_from_factions = {
	every_pop_faction = {
		
		if = {
			limit = {
				OR = {
					
					is_pop_faction_type = imperialist
					
					is_pop_faction_type = progressive
					
					
				}
				
			}
		
			export_trigger_value_to_variable = { trigger = support variable = populationControlsNotAllowed_support }
			multiply_variable = {which = populationControlsNotAllowed_support value = 100}
			# populationControlsNotAllowed_support is 0 to 100
			
			export_trigger_value_to_variable = { trigger = faction_approval variable = populationControlsNotAllowed_approval }
			# populationControlsNotAllowed_approval is 0 to 1

			ROOT = {
				change_variable = { which = $SUPPORT$ value = PREV.populationControlsNotAllowed_support }
				
				set_variable = { which = populationControlsNotAllowed_radicalism value = 1 }
				subtract_variable = { which = populationControlsNotAllowed_radicalism value = PREV.populationControlsNotAllowed_approval }
				
				multiply_variable = { which = populationControlsNotAllowed_radicalism value = PREV.populationControlsNotAllowed_support }

				change_variable = { which = $RADICALISM$ value = populationControlsNotAllowed_radicalism }
			}
		}
	}
}

calculate_populationControlsNotAllowed_movement_support_and_radical_from_numbers = {
	set_variable = { which = populationControlsNotAllowed_population value = 0}
	set_variable = { which = populationControlsNotAllowed_supporting_population value = 0}
	set_variable = { which = populationControlsNotAllowed_radical_population value = 0}
	
	every_owned_pop = {
		ROOT = {
			change_variable = { which = populationControlsNotAllowed_population value = 1}
		}
		
		if = {
			limit = {
				OR = {
					
					has_ethic = ethic_egalitarian
					has_ethic = ethic_fanatic_egalitarian
					
					has_ethic = ethic_militarist
					has_ethic = ethic_fanatic_militarist
					
				}
			}
			# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
			export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
			ROOT = {
				set_variable = {which = pop_populationControlsNotAllowed_radicalism value = 1}
				subtract_variable = { which = pop_populationControlsNotAllowed_radicalism value = PREV.pop_happiness }

				change_variable = { which = populationControlsNotAllowed_supporting_population value = 1 }
				change_variable = { which = populationControlsNotAllowed_radical_population value = pop_populationControlsNotAllowed_radicalism }
			}
		}
	}
	
	set_variable = { which = $SUPPORT$ value = populationControlsNotAllowed_supporting_population}
	divide_variable = { which = $SUPPORT$ value = populationControlsNotAllowed_population }
	multiply_variable = {which = $SUPPORT$ value = 100}

	set_variable = { which = $RADICALISM$ value = populationControlsNotAllowed_radical_population}
	divide_variable = { which = $RADICALISM$ value = populationControlsNotAllowed_population }
	multiply_variable = {which = $RADICALISM$ value = 100}
}
