
calculate_aiServitude_movement_support_and_radicalism_from_factions = {
	every_pop_faction = {
		
		if = {
			limit = {
				OR = {
					
					is_pop_faction_type = technologist
					
					is_pop_faction_type = traditionalist
					
					
				}
				
			}
		
			export_trigger_value_to_variable = { trigger = support variable = aiServitude_support }
			multiply_variable = {which = aiServitude_support value = 100}
			# aiServitude_support is 0 to 100
			
			export_trigger_value_to_variable = { trigger = faction_approval variable = aiServitude_approval }
			# aiServitude_approval is 0 to 1

			ROOT = {
				change_variable = { which = $SUPPORT$ value = PREV.aiServitude_support }
				
				set_variable = { which = aiServitude_radicalism value = 1 }
				subtract_variable = { which = aiServitude_radicalism value = PREV.aiServitude_approval }
				
				multiply_variable = { which = aiServitude_radicalism value = PREV.aiServitude_support }

				change_variable = { which = $RADICALISM$ value = aiServitude_radicalism }
			}
		}
	}
}

calculate_aiServitude_movement_support_and_radical_from_numbers = {
	set_variable = { which = aiServitude_population value = 0}
	set_variable = { which = aiServitude_supporting_population value = 0}
	set_variable = { which = aiServitude_radical_population value = 0}
	
	every_owned_pop = {
		ROOT = {
			change_variable = { which = aiServitude_population value = 1}
		}
		
		if = {
			limit = {
				OR = {
					
					has_ethic = ethic_materialist
					has_ethic = ethic_fanatic_materialist
					
					has_ethic = ethic_spiritualist
					has_ethic = ethic_fanatic_spiritualist
					
				}
			}
			# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
			export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
			ROOT = {
				set_variable = {which = pop_aiServitude_radicalism value = 1}
				subtract_variable = { which = pop_aiServitude_radicalism value = PREV.pop_happiness }

				change_variable = { which = aiServitude_supporting_population value = 1 }
				change_variable = { which = aiServitude_radical_population value = pop_aiServitude_radicalism }
			}
		}
	}
	
	set_variable = { which = $SUPPORT$ value = aiServitude_supporting_population}
	divide_variable = { which = $SUPPORT$ value = aiServitude_population }
	multiply_variable = {which = $SUPPORT$ value = 100}

	set_variable = { which = $RADICALISM$ value = aiServitude_radical_population}
	divide_variable = { which = $RADICALISM$ value = aiServitude_population }
	multiply_variable = {which = $RADICALISM$ value = 100}
}
