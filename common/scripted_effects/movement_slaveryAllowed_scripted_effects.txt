
calculate_slaveryAllowed_movement_support_and_radicalism_from_factions = {
	every_pop_faction = {
		
		if = {
			limit = {
				OR = {
					
					is_pop_faction_type = xenoist
					
					is_pop_faction_type = progressive
					
					
				}
				
			}
		
			export_trigger_value_to_variable = { trigger = support variable = slaveryAllowed_support }
			multiply_variable = {which = slaveryAllowed_support value = 100}
			# slaveryAllowed_support is 0 to 100
			
			export_trigger_value_to_variable = { trigger = faction_approval variable = slaveryAllowed_approval }
			# slaveryAllowed_approval is 0 to 1

			ROOT = {
				change_variable = { which = $SUPPORT$ value = PREV.slaveryAllowed_support }
				
				set_variable = { which = slaveryAllowed_radicalism value = 1 }
				subtract_variable = { which = slaveryAllowed_radicalism value = PREV.slaveryAllowed_approval }
				
				multiply_variable = { which = slaveryAllowed_radicalism value = PREV.slaveryAllowed_support }

				change_variable = { which = $RADICALISM$ value = slaveryAllowed_radicalism }
			}
		}
	}
}

calculate_slaveryAllowed_movement_support_and_radical_from_numbers = {
	set_variable = { which = slaveryAllowed_population value = 0}
	set_variable = { which = slaveryAllowed_supporting_population value = 0}
	set_variable = { which = slaveryAllowed_radical_population value = 0}
	
	every_owned_pop = {
		ROOT = {
			change_variable = { which = slaveryAllowed_population value = 1}
		}
		
		if = {
			limit = {
				OR = {
					
					has_ethic = ethic_xenophile
					has_ethic = ethic_fanatic_xenophile
					
					has_ethic = ethic_pacifist
					has_ethic = ethic_fanatic_pacifist
					
					has_ethic = ethic_egalitarian
					has_ethic = ethic_fanatic_egalitarian
					
				}
			}
			# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
			export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
			ROOT = {
				set_variable = {which = pop_slaveryAllowed_radicalism value = 1}
				subtract_variable = { which = pop_slaveryAllowed_radicalism value = PREV.pop_happiness }

				change_variable = { which = slaveryAllowed_supporting_population value = 1 }
				change_variable = { which = slaveryAllowed_radical_population value = pop_slaveryAllowed_radicalism }
			}
		}
	}
	
	set_variable = { which = $SUPPORT$ value = slaveryAllowed_supporting_population}
	divide_variable = { which = $SUPPORT$ value = slaveryAllowed_population }
	multiply_variable = {which = $SUPPORT$ value = 100}

	set_variable = { which = $RADICALISM$ value = slaveryAllowed_radical_population}
	divide_variable = { which = $RADICALISM$ value = slaveryAllowed_population }
	multiply_variable = {which = $RADICALISM$ value = 100}
}
