
calculate_defensiveWarsOnly_movement_support_and_radicalism_from_factions = {
	every_pop_faction = {
		
		if = {
			limit = {
				OR = {
					
					is_pop_faction_type = isolationist
					
					is_pop_faction_type = prosperity
					
					
				}
				
			}
		
			export_trigger_value_to_variable = { trigger = support variable = defensiveWarsOnly_support }
			multiply_variable = {which = defensiveWarsOnly_support value = 100}
			# defensiveWarsOnly_support is 0 to 100
			
			export_trigger_value_to_variable = { trigger = faction_approval variable = defensiveWarsOnly_approval }
			# defensiveWarsOnly_approval is 0 to 1

			ROOT = {
				change_variable = { which = $SUPPORT$ value = PREV.defensiveWarsOnly_support }
				
				set_variable = { which = defensiveWarsOnly_radicalism value = 1 }
				subtract_variable = { which = defensiveWarsOnly_radicalism value = PREV.defensiveWarsOnly_approval }
				
				multiply_variable = { which = defensiveWarsOnly_radicalism value = PREV.defensiveWarsOnly_support }

				change_variable = { which = $RADICALISM$ value = defensiveWarsOnly_radicalism }
			}
		}
	}
}

calculate_defensiveWarsOnly_movement_support_and_radical_from_numbers = {
	set_variable = { which = defensiveWarsOnly_population value = 0}
	set_variable = { which = defensiveWarsOnly_supporting_population value = 0}
	set_variable = { which = defensiveWarsOnly_radical_population value = 0}
	
	every_owned_pop = {
		ROOT = {
			change_variable = { which = defensiveWarsOnly_population value = 1}
		}
		
		if = {
			limit = {
				OR = {
					
					has_ethic = ethic_xenophobe
					has_ethic = ethic_fanatic_xenophobe
					
					has_ethic = ethic_pacifist
					has_ethic = ethic_fanatic_pacifist
					
				}
			}
			# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
			export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
			ROOT = {
				set_variable = {which = pop_defensiveWarsOnly_radicalism value = 1}
				subtract_variable = { which = pop_defensiveWarsOnly_radicalism value = PREV.pop_happiness }

				change_variable = { which = defensiveWarsOnly_supporting_population value = 1 }
				change_variable = { which = defensiveWarsOnly_radical_population value = pop_defensiveWarsOnly_radicalism }
			}
		}
	}
	
	set_variable = { which = $SUPPORT$ value = defensiveWarsOnly_supporting_population}
	divide_variable = { which = $SUPPORT$ value = defensiveWarsOnly_population }
	multiply_variable = {which = $SUPPORT$ value = 100}

	set_variable = { which = $RADICALISM$ value = defensiveWarsOnly_radical_population}
	divide_variable = { which = $RADICALISM$ value = defensiveWarsOnly_population }
	multiply_variable = {which = $RADICALISM$ value = 100}
}
