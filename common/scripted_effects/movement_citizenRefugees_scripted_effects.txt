
calculate_citizenRefugees_movement_support_and_radicalism_from_factions = {
	every_pop_faction = {
		
		if = {
			limit = {
				OR = {
					
					is_pop_faction_type = totalitarian
					
					is_pop_faction_type = imperialist
					
					
				}
				
			}
		
			export_trigger_value_to_variable = { trigger = support variable = citizenRefugees_support }
			multiply_variable = {which = citizenRefugees_support value = 100}
			# citizenRefugees_support is 0 to 100
			
			export_trigger_value_to_variable = { trigger = faction_approval variable = citizenRefugees_approval }
			# citizenRefugees_approval is 0 to 1

			ROOT = {
				change_variable = { which = $SUPPORT$ value = PREV.citizenRefugees_support }
				
				set_variable = { which = citizenRefugees_radicalism value = 1 }
				subtract_variable = { which = citizenRefugees_radicalism value = PREV.citizenRefugees_approval }
				
				multiply_variable = { which = citizenRefugees_radicalism value = PREV.citizenRefugees_support }

				change_variable = { which = $RADICALISM$ value = citizenRefugees_radicalism }
			}
		}
	}
}

calculate_citizenRefugees_movement_support_and_radical_from_numbers = {
	set_variable = { which = citizenRefugees_population value = 0}
	set_variable = { which = citizenRefugees_supporting_population value = 0}
	set_variable = { which = citizenRefugees_radical_population value = 0}
	
	every_owned_pop = {
		ROOT = {
			change_variable = { which = citizenRefugees_population value = 1}
		}
		
		if = {
			limit = {
				OR = {
					
					has_ethic = ethic_authoritarian
					has_ethic = ethic_fanatic_authoritarian
					
					has_ethic = ethic_militarist
					has_ethic = ethic_fanatic_militarist
					
				}
			}
			# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
			export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
			ROOT = {
				set_variable = {which = pop_citizenRefugees_radicalism value = 1}
				subtract_variable = { which = pop_citizenRefugees_radicalism value = PREV.pop_happiness }

				change_variable = { which = citizenRefugees_supporting_population value = 1 }
				change_variable = { which = citizenRefugees_radical_population value = pop_citizenRefugees_radicalism }
			}
		}
	}
	
	set_variable = { which = $SUPPORT$ value = citizenRefugees_supporting_population}
	divide_variable = { which = $SUPPORT$ value = citizenRefugees_population }
	multiply_variable = {which = $SUPPORT$ value = 100}

	set_variable = { which = $RADICALISM$ value = citizenRefugees_radical_population}
	divide_variable = { which = $RADICALISM$ value = citizenRefugees_population }
	multiply_variable = {which = $RADICALISM$ value = 100}
}
