
calculate_balanced_corporate_terms_movement_support_and_radicalism_from_factions = {
	every_pop_faction = {
		
		if = {
			limit = {
				OR = {
					
					
				}
				
				NAND = {
					is_pop_faction_type = ourBurden
					ROOT = { has_policy_flag = benevolent_terms }
				}
				
			}
		
			export_trigger_value_to_variable = { trigger = support variable = balanced_corporate_terms_support }
			multiply_variable = {which = balanced_corporate_terms_support value = 100}
			# balanced_corporate_terms_support is 0 to 100
			
			export_trigger_value_to_variable = { trigger = faction_approval variable = balanced_corporate_terms_approval }
			# balanced_corporate_terms_approval is 0 to 1

			ROOT = {
				change_variable = { which = $SUPPORT$ value = PREV.balanced_corporate_terms_support }
				
				set_variable = { which = balanced_corporate_terms_radicalism value = 1 }
				subtract_variable = { which = balanced_corporate_terms_radicalism value = PREV.balanced_corporate_terms_approval }
				
				multiply_variable = { which = balanced_corporate_terms_radicalism value = PREV.balanced_corporate_terms_support }

				change_variable = { which = $RADICALISM$ value = balanced_corporate_terms_radicalism }
			}
		}
	}
}

calculate_balanced_corporate_terms_movement_support_and_radical_from_numbers = {
	set_variable = { which = balanced_corporate_terms_population value = 0}
	set_variable = { which = balanced_corporate_terms_supporting_population value = 0}
	set_variable = { which = balanced_corporate_terms_radical_population value = 0}
	
	every_owned_pop = {
		ROOT = {
			change_variable = { which = balanced_corporate_terms_population value = 1}
		}
		
		if = {
			limit = {
				OR = {
					
				}
			}
			# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
			export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
			ROOT = {
				set_variable = {which = pop_balanced_corporate_terms_radicalism value = 1}
				subtract_variable = { which = pop_balanced_corporate_terms_radicalism value = PREV.pop_happiness }

				change_variable = { which = balanced_corporate_terms_supporting_population value = 1 }
				change_variable = { which = balanced_corporate_terms_radical_population value = pop_balanced_corporate_terms_radicalism }
			}
		}
	}
	
	set_variable = { which = $SUPPORT$ value = balanced_corporate_terms_supporting_population}
	divide_variable = { which = $SUPPORT$ value = balanced_corporate_terms_population }
	multiply_variable = {which = $SUPPORT$ value = 100}

	set_variable = { which = $RADICALISM$ value = balanced_corporate_terms_radical_population}
	divide_variable = { which = $RADICALISM$ value = balanced_corporate_terms_population }
	multiply_variable = {which = $RADICALISM$ value = 100}
}

calculate_balanced_corporate_terms_movement_inertia = {
	set_variable = { which = movement_balanced_corporate_terms_faction_support value = 0 }
	set_variable = { which = movement_balanced_corporate_terms_faction_radicalism value = 0 }
	set_variable = { which = movement_balanced_corporate_terms_numerical_support value = 0 }
	set_variable = { which = movement_balanced_corporate_terms_numerical_radicalism value = 0 }

	calculate_balanced_corporate_terms_movement_support_and_radicalism_from_factions = {
		SUPPORT = movement_balanced_corporate_terms_faction_support
		RADICALISM = movement_balanced_corporate_terms_faction_radicalism
	}
	calculate_balanced_corporate_terms_movement_support_and_radical_from_numbers = {
		SUPPORT = movement_balanced_corporate_terms_numerical_support
		RADICALISM = movement_balanced_corporate_terms_numerical_radicalism
	}

	set_variable = { which = movement_balanced_corporate_terms_support value = movement_balanced_corporate_terms_faction_support }
	change_variable = { which = movement_balanced_corporate_terms_support value = movement_balanced_corporate_terms_numerical_support }
	divide_variable = { which = movement_balanced_corporate_terms_support value = 2 }

	set_variable = { which = movement_balanced_corporate_terms_radicalism value = movement_balanced_corporate_terms_faction_radicalism }
	change_variable = { which = movement_balanced_corporate_terms_radicalism value = movement_balanced_corporate_terms_numerical_radicalism }
	divide_variable = { which = movement_balanced_corporate_terms_radicalism value = 2 }

	set_variable = {which = $INERTIA$ value = movement_balanced_corporate_terms_support}
	multiply_variable = { which = $INERTIA$ value = movement_balanced_corporate_terms_radicalism }
}
