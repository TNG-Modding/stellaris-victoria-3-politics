
calculate_no_retreat_doctrine_movement_support_and_radicalism_from_factions = {
	every_pop_faction = {
		
		if = {
			limit = {
				OR = {
					
					
				}
				
				NAND = {
					is_pop_faction_type = armedIsolationism
					ROOT = { has_policy_flag = defense_in_depth_doctrine }
				}
				
			}
		
			export_trigger_value_to_variable = { trigger = support variable = no_retreat_doctrine_support }
			multiply_variable = {which = no_retreat_doctrine_support value = 100}
			# no_retreat_doctrine_support is 0 to 100
			
			export_trigger_value_to_variable = { trigger = faction_approval variable = no_retreat_doctrine_approval }
			# no_retreat_doctrine_approval is 0 to 1

			ROOT = {
				change_variable = { which = $SUPPORT$ value = PREV.no_retreat_doctrine_support }
				
				set_variable = { which = no_retreat_doctrine_radicalism value = 1 }
				subtract_variable = { which = no_retreat_doctrine_radicalism value = PREV.no_retreat_doctrine_approval }
				
				multiply_variable = { which = no_retreat_doctrine_radicalism value = PREV.no_retreat_doctrine_support }

				change_variable = { which = $RADICALISM$ value = no_retreat_doctrine_radicalism }
			}
		}
	}
}

calculate_no_retreat_doctrine_movement_support_and_radical_from_numbers = {
	set_variable = { which = no_retreat_doctrine_population value = 0}
	set_variable = { which = no_retreat_doctrine_supporting_population value = 0}
	set_variable = { which = no_retreat_doctrine_radical_population value = 0}
	
	every_owned_pop = {
		ROOT = {
			change_variable = { which = no_retreat_doctrine_population value = 1}
		}
		
		if = {
			limit = {
				OR = {
					
				}
			}
			# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
			export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
			ROOT = {
				set_variable = {which = pop_no_retreat_doctrine_radicalism value = 1}
				subtract_variable = { which = pop_no_retreat_doctrine_radicalism value = PREV.pop_happiness }

				change_variable = { which = no_retreat_doctrine_supporting_population value = 1 }
				change_variable = { which = no_retreat_doctrine_radical_population value = pop_no_retreat_doctrine_radicalism }
			}
		}
	}
	
	set_variable = { which = $SUPPORT$ value = no_retreat_doctrine_supporting_population}
	divide_variable = { which = $SUPPORT$ value = no_retreat_doctrine_population }
	multiply_variable = {which = $SUPPORT$ value = 100}

	set_variable = { which = $RADICALISM$ value = no_retreat_doctrine_radical_population}
	divide_variable = { which = $RADICALISM$ value = no_retreat_doctrine_population }
	multiply_variable = {which = $RADICALISM$ value = 100}
}
