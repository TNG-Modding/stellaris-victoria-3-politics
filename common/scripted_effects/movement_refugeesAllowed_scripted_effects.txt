
calculate_refugeesAllowed_movement_support_and_radicalism_from_factions = {
	every_pop_faction = {
		
		if = {
			limit = {
				OR = {
					
					is_pop_faction_type = totalitarian
					
					is_pop_faction_type = xenoist
					
					
				}
				
			}
		
			export_trigger_value_to_variable = { trigger = support variable = refugeesAllowed_support }
			multiply_variable = {which = refugeesAllowed_support value = 100}
			# refugeesAllowed_support is 0 to 100
			
			export_trigger_value_to_variable = { trigger = faction_approval variable = refugeesAllowed_approval }
			# refugeesAllowed_approval is 0 to 1

			ROOT = {
				change_variable = { which = $SUPPORT$ value = PREV.refugeesAllowed_support }
				
				set_variable = { which = refugeesAllowed_radicalism value = 1 }
				subtract_variable = { which = refugeesAllowed_radicalism value = PREV.refugeesAllowed_approval }
				
				multiply_variable = { which = refugeesAllowed_radicalism value = PREV.refugeesAllowed_support }

				change_variable = { which = $RADICALISM$ value = refugeesAllowed_radicalism }
			}
		}
	}
}

calculate_refugeesAllowed_movement_support_and_radical_from_numbers = {
	set_variable = { which = refugeesAllowed_population value = 0}
	set_variable = { which = refugeesAllowed_supporting_population value = 0}
	set_variable = { which = refugeesAllowed_radical_population value = 0}
	
	every_owned_pop = {
		ROOT = {
			change_variable = { which = refugeesAllowed_population value = 1}
		}
		
		if = {
			limit = {
				OR = {
					
					has_ethic = ethic_authoritarian
					has_ethic = ethic_fanatic_authoritarian
					
					has_ethic = ethic_xenophile
					has_ethic = ethic_fanatic_xenophile
					
				}
			}
			# A perfectly happy pop has happiness 1, and unhappy 0. An perfectly unhappy pop in thise case provides 1 radicalism support whereas his happy cousin contributes none.
			export_modifier_to_variable = { modifier = pop_happiness variable = pop_happiness }
			ROOT = {
				set_variable = {which = pop_refugeesAllowed_radicalism value = 1}
				subtract_variable = { which = pop_refugeesAllowed_radicalism value = PREV.pop_happiness }

				change_variable = { which = refugeesAllowed_supporting_population value = 1 }
				change_variable = { which = refugeesAllowed_radical_population value = pop_refugeesAllowed_radicalism }
			}
		}
	}
	
	set_variable = { which = $SUPPORT$ value = refugeesAllowed_supporting_population}
	divide_variable = { which = $SUPPORT$ value = refugeesAllowed_population }
	multiply_variable = {which = $SUPPORT$ value = 100}

	set_variable = { which = $RADICALISM$ value = refugeesAllowed_radical_population}
	divide_variable = { which = $RADICALISM$ value = refugeesAllowed_population }
	multiply_variable = {which = $RADICALISM$ value = 100}
}
